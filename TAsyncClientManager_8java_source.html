63  private class SelectThread extends Thread {
64  private final Selector selector;
65  private volatile boolean running;
66  private final TreeSet<TAsyncMethodCall> timeoutWatchSet =
67  new TreeSet<TAsyncMethodCall>(new TAsyncMethodCallTimeoutComparator());
68 
69  public SelectThread() throws IOException {
70  this.selector = SelectorProvider.provider().openSelector();
71  this.running = true;
72  this.setName("TAsyncClientManager#SelectorThread " + this.getId());
73 
74  // We don't want to hold up the JVM when shutting down
75  setDaemon(true);
76  }
77 
78  public Selector getSelector() {
79  return selector;
80  }
81 
82  public void finish() {
83  running = false;
84  selector.wakeup();
85  }

87  public void run() {
88  while (running) {
89  try {
90  try {
91  if (timeoutWatchSet.size() == 0) {
92  // No timeouts, so select indefinitely
93  selector.select();
94  } else {
95  // We have a timeout pending, so calculate the time until then and select
96  // appropriately
97  long nextTimeout = timeoutWatchSet.first().getTimeoutTimestamp();
98  long selectTime = nextTimeout - System.currentTimeMillis();
99  if (selectTime > 0) {
100  // Next timeout is in the future, select and wake up then
101  selector.select(selectTime);
102  } else {
103  // Next timeout is now or in past, select immediately so we can time out
104  selector.selectNow();
105  }
106  }
107  } catch (IOException e) {
108  LOGGER.error("Caught IOException in TAsyncClientManager!", e);
109  }
110  transitionMethods();
111  timeoutMethods();
112  startPendingMethods();
113  } catch (Exception exception) {
114  LOGGER.error("Ignoring uncaught exception in SelectThread", exception);
115  }
116  }
117  }
118 
119  // Transition methods for ready keys
120  private void transitionMethods() {
121  try {
122  Iterator<SelectionKey> keys = selector.selectedKeys().iterator();
123  while (keys.hasNext()) {
124  SelectionKey key = keys.next();
125  keys.remove();
126  if (!key.isValid()) {
127  // this can happen if the method call experienced an error and the
128  // key was cancelled. can also happen if we timeout a method, which
129  // results in a channel close.
130  // just skip
131  continue;
132  }
133  TAsyncMethodCall methodCall = (TAsyncMethodCall) key.attachment();
134  methodCall.transition(key);
135 
136  // If done or error occurred, remove from timeout watch set
137  if (methodCall.isFinished() || methodCall.getClient().hasError()) {
138  timeoutWatchSet.remove(methodCall);
139  }
140  }
141  } catch (ClosedSelectorException e) {
142  LOGGER.error("Caught ClosedSelectorException in TAsyncClientManager!", e);
143  }
144  }
145 
146  // Timeout any existing method calls
147  private void timeoutMethods() {
148  Iterator<TAsyncMethodCall> iterator = timeoutWatchSet.iterator();
149  long currentTime = System.currentTimeMillis();
150  while (iterator.hasNext()) {
151  TAsyncMethodCall methodCall = iterator.next();
152  if (currentTime >= methodCall.getTimeoutTimestamp()) {
153  iterator.remove();
154  methodCall.onError(
155  new TimeoutException(
156  "Operation "
157  + methodCall.getClass()
158  + " timed out after "
159  + (currentTime - methodCall.getStartTime())
160  + " ms."));
161  } else {
162  break;
163  }
164  }
165  }
166 
167  // Start any new calls
168  private void startPendingMethods() {
169  TAsyncMethodCall methodCall;
170  while ((methodCall = pendingCalls.poll()) != null) {
171  // Catch registration errors. method will catch transition errors and cleanup.
172  try {
173  methodCall.start(selector);
174 
175  // If timeout specified and first transition went smoothly, add to timeout watch set
176  TAsyncClient client = methodCall.getClient();
177  if (client.hasTimeout() && !client.hasError()) {
178  timeoutWatchSet.add(methodCall);
179  }
180  } catch (Exception exception) {
181  LOGGER.warn("Caught exception in TAsyncClientManager!", exception);
182  methodCall.onError(exception);
183  }
184  }
185  }
186  }

189  private static class TAsyncMethodCallTimeoutComparator implements Comparator<TAsyncMethodCall> {
190  public int compare(TAsyncMethodCall left, TAsyncMethodCall right) {
191  if (left.getTimeoutTimestamp() == right.getTimeoutTimestamp()) {
192  return (int) (left.getSequenceId() - right.getSequenceId());
193  } else {
194  return (int) (left.getTimeoutTimestamp() - right.getTimeoutTimestamp());
195  }
196  }
197  }
