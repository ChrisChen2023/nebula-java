101  stopped_ = false;
102  try {
103  serverTransport_.listen();
104  } catch (TTransportException ttx) {
105  LOGGER.error("Error occurred during listening.", ttx);
106  return;
107  }
108 
109  while (!stopped_) {
110  TTransport client = null;
111  TProcessor processor = null;
112  TTransport inputTransport = null;
113  TTransport outputTransport = null;
114  TProtocol inputProtocol = null;
115  TProtocol outputProtocol = null;
116  try {
117  client = serverTransport_.accept();
118  if (client != null) {
119  processor = processorFactory_.getProcessor(client);
120  inputTransport = inputTransportFactory_.getTransport(client);
121  inputProtocol = inputProtocolFactory_.getProtocol(inputTransport);
122  // THeaderProtocol must be the same instance for both input and output
123  if (inputProtocol instanceof THeaderProtocol) {
124  outputProtocol = inputProtocol;
125  } else {
126  outputTransport = outputTransportFactory_.getTransport(client);
127  outputProtocol = outputProtocolFactory_.getProtocol(outputTransport);
128  }
129  TRpcConnectionContext server_ctx =
130  new TRpcConnectionContext(client, inputProtocol, outputProtocol);
131  while (processor.process(inputProtocol, outputProtocol, server_ctx)) {}
132  }
133  } catch (TTransportException ttx) {
134  // Client died, just move on
135  } catch (TException tx) {
136  if (!stopped_) {
137  LOGGER.error("Thrift error occurred during processing of message.", tx);
138  }
139  } catch (Exception x) {
140  if (!stopped_) {
141  LOGGER.error("Error occurred during processing of message.", x);
142  }
143  }
144 
145  if (inputTransport != null) {
146  inputTransport.close();
147  }
148 
149  if (outputTransport != null) {
150  outputTransport.close();
151  }
152  }
153  }
