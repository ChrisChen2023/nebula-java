280  protected void readFrame(int reqLen) throws TTransportException {
281  transport_.readAll(i32buf, 0, 4);
282  int word1 = decodeWord(i32buf);
283 
284  if ((word1 & TBinaryProtocol.VERSION_MASK) == TBinaryProtocol.VERSION_1) {
285  throw new THeaderException("This transport does not support Unframed");
286  } else if (word1 == HTTP_SERVER_MAGIC) {
287  throw new THeaderException("This transport does not support HTTP");
288  } else {
289  if (word1 - 4 > MAX_FRAME_SIZE) {
290  // special case for the most common question in user-group
291  // this will probably saves hours of engineering effort.
292  int magic1 = 0x61702048; // ASCII "ap H" in little endian
293  int magic2 = 0x6C6C6F63; // ASCII "lloc" in little endian
294  if (word1 == magic1 || word1 == magic2) {
295  throw new TTransportException(
296  "The Thrift server received an ASCII request and safely ignored it. "
297  + "In all likelihood, this isn't the reason of your problem "
298  + "(probably a local daemon sending HTTP content to all listening ports).");
299  }
300  throw new TTransportException("Framed transport frame " + "is too large");
301  }
302 
303  // Could be framed or header format. Check next word.
304  transport_.readAll(i32buf, 0, 4);
305  int version = decodeWord(i32buf);
306  if ((version & TBinaryProtocol.VERSION_MASK) == TBinaryProtocol.VERSION_1) {
307  clientType = ClientTypes.FRAMED_DEPRECATED;
308  byte[] buff = new byte[word1];
309  System.arraycopy(i32buf, 0, buff, 0, 4);
310  transport_.readAll(buff, 4, word1 - 4);
311  readBuffer_.reset(buff);
312  } else if ((version & HEADER_MAGIC_MASK) == HEADER_MAGIC) {
313  clientType = ClientTypes.HEADERS;
314  if (word1 - 4 < 10) {
315  throw new TTransportException("Header transport frame " + "is too small");
316  }
317  byte[] buff = new byte[word1];
318  System.arraycopy(i32buf, 0, buff, 0, 4);
319 
320  // read packet minus version
321  transport_.readAll(buff, 4, word1 - 4);
322  flags = version & HEADER_FLAGS_MASK;
323  // read seqId
324  seqId = decodeWord(buff, 4);
325  int headerSize = decodeShort(buff, 8);
326 
327  readHeaderFormat(headerSize, buff);
328  } else {
329  clientType = ClientTypes.UNKNOWN;
330  throw new THeaderException("Unsupported client type");
331  }
332  }
333  }

341  private int readVarint32Buf(ByteBuffer frame) {
342  int result = 0;
343  int shift = 0;
344 
345  while (true) {
346  byte b = frame.get();
347  result |= (int) (b & 0x7f) << shift;
348  if ((b & 0x80) != 0x80) {
349  break;
350  }
351  shift += 7;
352  }
353 
354  return result;
355  }

357  private void writeVarint(ByteBuffer out, int n) {
358  while (true) {
359  if ((n & ~0x7F) == 0) {
360  out.put((byte) n);
361  break;
362  } else {
363  out.put((byte) (n | 0x80));
364  n >>>= 7;
365  }
366  }
367  }

382  private void readHeaderFormat(int headerSize, byte[] buff) throws TTransportException {
383  ByteBuffer frame = ByteBuffer.wrap(buff);
384  frame.position(10); // Advance past version, flags, seqid
385 
386  headerSize = headerSize * 4;
387  int endHeader = headerSize + frame.position();
388  if (headerSize > frame.remaining()) {
389  throw new TTransportException("Header size is larger than frame");
390  }
391  protoId = readVarint32Buf(frame);
392  int numTransforms = readVarint32Buf(frame);
393 
394  // Clear out any previous transforms
395  readTransforms = new ArrayList<Integer>(numTransforms);
396 
397  if (protoId == T_JSON_PROTOCOL && clientType != ClientTypes.HTTP) {
398  throw new TTransportException("Trying to recv JSON encoding " + "over binary");
399  }
400 
401  // Read in the headers. Data for each varies. See
402  // doc/HeaderFormat.txt
403  int hmacSz = 0;
404  for (int i = 0; i < numTransforms; i++) {
405  int transId = readVarint32Buf(frame);
406  if (transId == Transforms.ZLIB_TRANSFORM.getValue()) {
407  readTransforms.add(transId);
408  } else if (transId == Transforms.SNAPPY_TRANSFORM.getValue()) {
409  throw new THeaderException("Snappy transform no longer supported");
410  } else if (transId == Transforms.HMAC_TRANSFORM.getValue()) {
411  throw new THeaderException("Hmac transform no longer supported");
412  } else {
413  throw new THeaderException("Unknown transform during recv");
414  }
415  }
416 
417  // Read the info section.
418  readHeaders.clear();
419  while (frame.position() < endHeader) {
420  int infoId = readVarint32Buf(frame);
421  if (infoId == Infos.INFO_KEYVALUE.getValue()) {
422  int numKeys = readVarint32Buf(frame);
423  for (int i = 0; i < numKeys; i++) {
424  String key = readString(frame);
425  String value = readString(frame);
426  readHeaders.put(key, value);
427  }
428  } else if (infoId == Infos.INFO_PKEYVALUE.getValue()) {
429  int numKeys = readVarint32Buf(frame);
430  for (int i = 0; i < numKeys; i++) {
431  String key = readString(frame);
432  String value = readString(frame);
433  readPersistentHeaders.put(key, value);
434  }
435  } else {
436  // Unknown info ID, continue on to reading data.
437  break;
438  }
439  }
440  readHeaders.putAll(readPersistentHeaders);
441 
442  // Read in the data section.
443  frame.position(endHeader);
444  frame.limit(frame.limit() - hmacSz); // limit to data without mac
445 
446  frame = untransform(frame);
447  readBuffer_.reset(frame.array(), frame.position(), frame.remaining());
448  }

450  private ByteBuffer untransform(ByteBuffer data) throws TTransportException {
451 
452  if (readTransforms.contains(Transforms.ZLIB_TRANSFORM.getValue())) {
453  try {
454  Inflater decompressor = new Inflater();
455  decompressor.setInput(data.array(), data.position(), data.remaining());
456  int length = 0;
457  ArrayList<byte[]> outBytes = new ArrayList<byte[]>();
458  while (!decompressor.finished()) {
459  byte[] output = new byte[zlibBufferSize];
460  length += decompressor.inflate(output);
461  outBytes.add(output);
462  }
463  decompressor.end();
464 
465  // Ugh output wants to be a list of blocks, we just want a buffer
466  if (outBytes.size() == 1) {
467  data = ByteBuffer.wrap(outBytes.get(0));
468  } else {
469  ByteBuffer output = ByteBuffer.allocate(length);
470  for (byte[] outBlock : outBytes) {
471  output.put(outBlock, 0, Math.min(zlibBufferSize, length));
472  length -= outBlock.length;
473  }
474  data = output;
475  data.position(0);
476  }
477  } catch (DataFormatException dfe) {
478  throw new THeaderException("Could not inflate data");
479  }
480  if (!writeTransforms.contains(Transforms.ZLIB_TRANSFORM)) {
481  writeTransforms.add(Transforms.ZLIB_TRANSFORM);
482  }
483  }
484  return data;
485  }

487  private ByteBuffer transform(ByteBuffer data) throws TTransportException {
488 
489  if (writeTransforms.contains(Transforms.ZLIB_TRANSFORM)) {
490  byte[] output = new byte[data.limit() + 512]; // output might be larger
491  Deflater compressor = new Deflater();
492  try {
493  compressor.setInput(data.array(), data.position(), data.remaining());
494  compressor.finish();
495  int length = compressor.deflate(output);
496  if (!compressor.finished()) {
497  // Output buffer was not big enough. Unlikely.
498  // If you hit this, you probably shouldn't be using ZLIB_TRANSFORM :)
499  throw new TTransportException("Output compress buffer not big enough");
500  }
501  data = ByteBuffer.wrap(output);
502  data.limit(length);
503  } finally {
504  compressor.end();
505  }
506  }
507 
508  return data;
509  }

511  private int getWriteHeadersSize(Map<String, String> headers) {
512  if (headers.size() == 0) {
513  return 0;
514  }
515 
516  int len = 10; // 5 bytes varint for info header type
517  // 5 bytes varint for info headers count
518  for (Map.Entry<String, String> header : headers.entrySet()) {
519  len += 10; // 5 bytes varint for key size and
520  // 5 bytes varint for value size
521  len += header.getKey().length();
522  len += header.getValue().length();
523  }
524  return len;
525  }

527  private ByteBuffer flushInfoHeaders(Infos info, Map<String, String> headers) {
528  ByteBuffer infoData = ByteBuffer.allocate(getWriteHeadersSize(headers));
529  if (!headers.isEmpty()) {
530  writeVarint(infoData, info.getValue());
531  writeVarint(infoData, headers.size());
532  for (Map.Entry<String, String> pairs : headers.entrySet()) {
533  writeString(infoData, pairs.getKey());
534  writeString(infoData, pairs.getValue());
535  }
536  headers.clear();
537  }
538  infoData.limit(infoData.position());
539  infoData.position(0);
540  return infoData;
541  }

556  public void flushImpl(boolean oneway) throws TTransportException {
557  try {
558  // Check if this is a TApplicationException
559  TApplicationException tae = null;
560  byte[] buf = writeBuffer_.get();
561  int len = writeBuffer_.len();
562  if (len >= 2
563  && buf[0] == TCompactProtocol.PROTOCOL_ID
564  && ((buf[1] >> TCompactProtocol.TYPE_SHIFT_AMOUNT) & 0x03) == TMessageType.EXCEPTION) {
565  // Compact
566  TCompactProtocol proto = new TCompactProtocol(new TMemoryInputTransport(buf));
567  @SuppressWarnings("unused")
568  TMessage msg = proto.readMessageBegin();
569  tae = TApplicationException.read(proto);
570  } else if (len >= 4
571  && ((buf[0] << 24) | (buf[1] << 16)) == TBinaryProtocol.VERSION_1
572  && buf[3] == TMessageType.EXCEPTION) {
573  // Binary
574  TBinaryProtocol proto = new TBinaryProtocol(new TMemoryInputTransport(buf));
575  @SuppressWarnings("unused")
576  TMessage msg = proto.readMessageBegin();
577  tae = TApplicationException.read(proto);
578  }
579 
580  if (tae != null) {
581  if (!writeHeaders.containsKey("uex")) {
582  writeHeaders.put("uex", "TApplicationException");
583  }
584  if (!writeHeaders.containsKey("uexw")) {
585  writeHeaders.put("uexw", tae.getMessage() == null ? "[null]" : tae.getMessage());
586  }
587  }
588  } catch (TException e) {
589  // Failed parsing a TApplicationException, so don't write headers
590  }
591 
592  ByteBuffer frame = ByteBuffer.wrap(writeBuffer_.get());
593  frame.limit(writeBuffer_.len());
594  writeBuffer_.reset();
595  if (clientType == ClientTypes.HEADERS) {
596  frame = transform(frame);
597  }
598 
599  if (frame.remaining() > MAX_FRAME_SIZE) {
600  throw new TTransportException(
601  "Attempting to send frame that is "
602  + "too large: "
603  + Integer.toString(frame.remaining()));
604  }
605 
606  if (protoId == T_JSON_PROTOCOL && clientType != ClientTypes.HTTP) {
607  throw new TTransportException("Trying to send JSON encoding" + " over binary");
608  }
609 
610  if (clientType == ClientTypes.HEADERS) {
611 
612  // Each varint could be up to 5 in size.
613  ByteBuffer transformData = ByteBuffer.allocate(writeTransforms.size() * 5);
614 
615  // For now, no transforms require data.
616  int numTransforms = writeTransforms.size();
617  for (Transforms trans : writeTransforms) {
618  writeVarint(transformData, trans.getValue());
619  }
620  transformData.limit(transformData.position());
621  transformData.position(0);
622 
623  if (identity != null && identity.length() > 0) {
624  writeHeaders.put(ID_VERSION_HEADER, ID_VERSION);
625  writeHeaders.put(IDENTITY_HEADER, identity);
626  }
627 
628  ByteBuffer infoData1 = flushInfoHeaders(Infos.INFO_PKEYVALUE, writePersistentHeaders);
629  ByteBuffer infoData2 = flushInfoHeaders(Infos.INFO_KEYVALUE, writeHeaders);
630 
631  ByteBuffer headerData = ByteBuffer.allocate(10);
632  writeVarint(headerData, protoId);
633  writeVarint(headerData, numTransforms);
634  headerData.limit(headerData.position());
635  headerData.position(0);
636 
637  int headerSize =
638  transformData.remaining()
639  + infoData1.remaining()
640  + infoData2.remaining()
641  + headerData.remaining();
642  int paddingSize = 4 - headerSize % 4;
643  headerSize += paddingSize;
644 
645  // Allocate buffer for the headers.
646  // 14 bytes for sz, magic , flags , seqId , headerSize
647  ByteBuffer out = ByteBuffer.allocate(headerSize + 14);
648 
649  // See thrift/doc/HeaderFormat.txt for more info on wire format
650  encodeInt(out, 10 + headerSize + frame.remaining());
651  encodeShort(out, HEADER_MAGIC >> 16);
652  encodeShort(out, flags);
653  encodeInt(out, seqId);
654  encodeShort(out, headerSize / 4);
655 
656  out.put(headerData);
657  out.put(transformData);
658  out.put(infoData1);
659  out.put(infoData2);
660 
661  // There are no info headers for this version
662  // Pad out the header with 0x00
663  for (int i = 0; i < paddingSize; i++) {
664  out.put((byte) 0x00);
665  }
666  out.position(0);
667 
668  transport_.write(out.array(), out.position(), out.remaining());
669  transport_.write(frame.array(), frame.position(), frame.remaining());
670  } else if (clientType == ClientTypes.FRAMED_DEPRECATED) {
671  ByteBuffer out = ByteBuffer.allocate(4);
672  encodeInt(out, frame.remaining());
673  out.position(0);
674  transport_.write(out.array(), out.position(), out.remaining());
675  transport_.write(frame.array(), frame.position(), frame.remaining());
676  } else if (clientType == ClientTypes.HTTP) {
677  throw new TTransportException("HTTP is unimplemented in this language");
678  } else {
679  throw new TTransportException("Unknown client type on send");
680  }
681 
682  if (oneway) {
683  transport_.onewayFlush();
684  } else {
685  transport_.flush();
686  }
687  }
