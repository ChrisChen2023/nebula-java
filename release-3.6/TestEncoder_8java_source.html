113  public int getSpaceVidLen(String spaceName) throws RuntimeException {
114  SpaceItem spaceItem = cacheImplTest.getSpace(spaceName);
115  if (spaceItem == null) {
116  throw new RuntimeException("SpaceName: " + spaceName + "is not existed");
117  }
118  if (spaceItem.properties.vid_type.type != PropertyType.FIXED_STRING) {
119  throw new RuntimeException("Only supported fixed string vid type.");

122  }

132  public int getPartId(int partNum, String vertexId) {
133  long hash = MurmurHash2.hash64(vertexId.getBytes(), vertexId.length(), 0xc70f6907);
134  long hashValue = Long.parseUnsignedLong(Long.toUnsignedString(hash));
135  return (int) (Math.floorMod(hashValue, partNum) + 1);
136  }
