414  private void select() {
415  try {
416  // wait for io events.
417  selector.select(TNonblockingServer.selectWaitTime_);
418 
419  // process the io events we received
420  Iterator<SelectionKey> selectedKeys = selector.selectedKeys().iterator();
421  while (!stopped_ && selectedKeys.hasNext()) {
422  SelectionKey key = selectedKeys.next();
423  selectedKeys.remove();
424 
425  // skip if not valid
426  if (!key.isValid()) {
427  cleanupSelectionkey(key);
428  continue;
429  }
430 
431  // if the key is marked Accept, then it has to be the server
432  // transport.
433  if (key.isAcceptable()) {
434  handleAccept();
435  } else if (key.isReadable()) {
436  // deal with reads
437  key.interestOps(0);
438  TcpHandler readHandler = new TcpReader(key);
439  try {
440  TNonblockingServer.this.submitTask(readHandler);
441  } catch (ServerOverloadedException e) {
442  cleanupSelectionkey(key);
443  }
444  } else if (key.isWritable()) {
445  // deal with writes
446  key.interestOps(0);
447  TcpHandler writeHandler = new TcpWriter(key);
448  try {
449  TNonblockingServer.this.submitTask(writeHandler);
450  } catch (ServerOverloadedException e) {
451  cleanupSelectionkey(key);
452  }
453  } else {
454  LOGGER.warn("Unexpected state in select! " + key.interestOps());
455  }
456  }
457  } catch (IOException e) {
458  LOGGER.warn("Got an IOException while selecting!", e);
459  }
460  }

463  synchronized (cancellations) {

466  }
467  cancellations.clear();

473  TNonblockingTransport client = null;

492  private void handleRead(SelectionKey key) {
493  FrameBuffer buffer = (FrameBuffer) key.attachment();
494  if (buffer.read()) {
495  // if the buffer's frame read is complete, invoke the method.
496  if (buffer.isFrameFullyRead()) {
497  try {
498  requestInvoke(buffer);
499  } catch (ServerOverloadedException e) {
500  TNonblockingServer.this.requestCancellation(buffer);
501  }
502  } else {
503  buffer.changeSelectInterests();
504  }
505  } else {
506  TNonblockingServer.this.requestCancellation(buffer);
507  }
508  }
509 
511  private void handleWrite(SelectionKey key) {
512  FrameBuffer buffer = (FrameBuffer) key.attachment();
513  if (buffer.write()) {
514  buffer.changeSelectInterests();
515  } else {
516  TNonblockingServer.this.requestCancellation(buffer);
517  }
518  }

539  protected class FrameBuffer {
540  //
541  // Possible states for the FrameBuffer state machine.
542  //
543  // in the midst of reading the frame size off the wire
544  private static final int READING_FRAME_SIZE = 1;
545  // reading the actual frame data now, but not all the way done yet
546  private static final int READING_FRAME = 2;
547  // completely read the frame, so an invocation can now happen
548  private static final int READ_FRAME_COMPLETE = 3;
549  // waiting to get switched to listening for write events
550  private static final int AWAITING_REGISTER_WRITE = 4;
551  // started writing response data, not fully complete yet
552  private static final int WRITING = 6;
553  // another thread wants this framebuffer to go back to reading
554  private static final int AWAITING_REGISTER_READ = 7;
555  // we want our transport and selection key invalidated in the selector
556  // thread
557  private static final int AWAITING_CLOSE = 8;
558 
559  //
560  // Instance variables
561  //
562 
563  private final long creationTime_;
564  // the actual transport hooked up to the client.
565  private final TNonblockingTransport trans_;
566 
567  // the SelectionKey that corresponds to our transport
568  private final SelectionKey selectionKey_;
569 
570  // where in the process of reading/writing are we?
571  private int state_ = READING_FRAME_SIZE;
572 
573  // the ByteBuffer we'll be using to write and read, depending on the state
574  private ByteBuffer buffer_;
575 
576  private TByteArrayOutputStream response_;
577 
578  public FrameBuffer(final TNonblockingTransport trans, final SelectionKey selectionKey) {
579  creationTime_ = System.currentTimeMillis();
580  trans_ = trans;
581  selectionKey_ = selectionKey;
582  buffer_ = ByteBuffer.allocate(4);
583  }

588  }

596  public synchronized boolean read() {
597  if (state_ == READING_FRAME_SIZE) {
598  // try to read the frame size completely
599  if (!internalRead()) {
600  return false;
601  }
602 
603  // if the frame size has been read completely, then prepare to read the
604  // actual frame.
605  if (buffer_.remaining() == 0) {
606  // pull out the frame size as an integer.
607  int frameSize = buffer_.getInt(0);
608  if (frameSize <= 0) {
609  LOGGER.error(
610  "Read an invalid frame size of "
611  + frameSize
612  + ". Are you using TFramedTransport on the client side?");
613  return false;
614  }
615 
616  // if this frame will always be too large for this server, log the
617  // error and close the connection.
618  int newBufferSize = frameSize + 4;
619  if (newBufferSize > MAX_READ_BUFFER_BYTES) {
620  LOGGER.error(
621  "Read a frame size of "
622  + frameSize
623  + ", which is bigger than the maximum allowable buffer size "
624  + "for ALL connections.");
625  return false;
626  }
627 
628  // increment the amount of memory allocated to read buffers
629  // if this frame will push us over the memory limit, then return.
630  // with luck, more memory will free up the next time around.
631  if (readBufferBytesAllocated.addAndGet(newBufferSize) > MAX_READ_BUFFER_BYTES) {
632  readBufferBytesAllocated.addAndGet(-newBufferSize);
633  return true;
634  }
635  // reallocate the readbuffer as a frame-sized buffer
636  buffer_ = ByteBuffer.allocate(newBufferSize);
637  // put the frame size at the head of the buffer
638  buffer_.putInt(frameSize);
639 
640  state_ = READING_FRAME;
641  } else {
642  // this skips the check of READING_FRAME state below, since we can't
643  // possibly go on to that state if there's data left to be read at
644  // this one.
645  return true;
646  }
647  }
648 
649  // it is possible to fall through from the READING_FRAME_SIZE section
650  // to READING_FRAME if there's already some frame data available once
651  // READING_FRAME_SIZE is complete.
652 
653  if (state_ == READING_FRAME) {
654  if (!internalRead()) {
655  return false;
656  }
657 
658  // since we're already in the select loop here for sure, we can just
659  // modify our selection key directly.
660  if (buffer_.remaining() == 0) {
661  state_ = READ_FRAME_COMPLETE;
662  }
663 
664  return true;
665  }
666 
667  // if we fall through to this point, then the state must be invalid.
668  LOGGER.error("Read was called but state is invalid (" + state_ + ")");
669 
670  return false;
671  }
672 
674  public synchronized boolean write() {
675  if (state_ == WRITING) {
676  try {
677  if (trans_.write(buffer_) < 0) {
678  return false;
679  }
680  } catch (IOException e) {
681  LOGGER.warn("Got an IOException during write!", e);
682  return false;
683  }
684 
685  // we're done writing. now we need to switch back to reading.
686  if (buffer_.remaining() == 0) {
687  state_ = AWAITING_REGISTER_READ;
688  }
689 
690  return true;
691  }
692 
693  LOGGER.error("Write was called, but state is invalid (" + state_ + ")");
694  return false;
695  }
696 
698  public synchronized void changeSelectInterests() {
699  if ((state_ == AWAITING_REGISTER_WRITE) || (state_ == WRITING)) {
700  state_ = WRITING;
701  // set the OP_WRITE interest. Again set it after the assigning of the
702  // state in the state machine to avoid inconsistencies between the
703  // state machine and the interests.
704  selectionKey_.interestOps(SelectionKey.OP_WRITE);
705  } else if (state_ == AWAITING_REGISTER_READ) {
706  // get ready for another go-around
707  buffer_ = ByteBuffer.allocate(4);
708  state_ = READING_FRAME_SIZE;
709  // Once you set the interest, the selector thread could schedule a read
710  // or write on a different thread before you can set the state machine
711  // state. That's why the state machine state *MUST* be set before
712  // setting the interest
713  selectionKey_.interestOps(SelectionKey.OP_READ);
714  } else if (state_ == AWAITING_CLOSE) {
715  cancel();
716  } else if ((state_ == READING_FRAME_SIZE) || (state_ == READING_FRAME)) {
717  // Register interest to continue reading.
718  selectionKey_.interestOps(SelectionKey.OP_READ);
719  } else {
720  LOGGER.error("changeSelectInterest was called, but state is invalid (" + state_ + ")");
721  }
722 
723  // All selector interest ops updates MUST be followed with a call to wake
724  // up the selector, unless this thread is currently the selector.
725  if (Thread.currentThread() != selectThread_) {
726  selectThread_.wakeupSelector();
727  }
728  }

736  public synchronized void close() {

738  // buffer that we need to subtract for our memory accounting.
739  if (state_ == READING_FRAME || state_ == READ_FRAME_COMPLETE) {
740  readBufferBytesAllocated.addAndGet(-buffer_.capacity());
741  }

756  private void responseReady() {
757  // the read buffer is definitely no longer in use, so we will decrement
758  // our read buffer count. we do this here as well as in close because
759  // we'd like to free this read memory up as quickly as possible for other
760  // clients.
761  readBufferBytesAllocated.addAndGet(-buffer_.capacity());
762 
763  if (response_.len() == 0) {
764  // go straight to reading again. this was probably an oneway method
765  state_ = AWAITING_REGISTER_READ;
766  buffer_ = null;
767  } else {
768  buffer_ = ByteBuffer.wrap(response_.get(), 0, response_.len());
769 
770  // set state that we're waiting to be switched to write. we do this
771  // asynchronously through requestSelectInterestChange() because
772  // there is a
773  // possibility that we're not in the main thread, and thus currently
774  // blocked in select(). (this functionality is in place for the sake of
775  // the HsHa server.)
776  state_ = AWAITING_REGISTER_WRITE;
777  }
778  changeSelectInterests();
779  }

782  public synchronized void invoke() {
783  TTransport inTrans;
784  TProtocol inProt;
785  TProtocol outProt;
786  // For THeader inProt and outProt must be the same object
787  if (inputProtocolFactory_ instanceof THeaderProtocol.Factory) {
788  inTrans = getInputOutputTransport();
789  inProt = inputProtocolFactory_.getProtocol(inTrans);
790  outProt = inProt;
791  } else {
792  inTrans = getInputTransport();
793  inProt = inputProtocolFactory_.getProtocol(inTrans);
794  outProt = outputProtocolFactory_.getProtocol(getOutputTransport());
795  }
796 
797  try {
798  TRpcConnectionContext server_ctx = new TRpcConnectionContext(trans_, inProt, outProt);
799  processorFactory_.getProcessor(inTrans).process(inProt, outProt, server_ctx);
800  responseReady();
801  return;
802  } catch (TException te) {
803  LOGGER.warn("Exception while invoking!", te);
804  } catch (Exception e) {
805  LOGGER.error("Unexpected exception while invoking!", e);
806  }
807 
808  // Exceptions will fall through to here.
809  // We can clear out the buffer here because it is no longer being used
810  readBufferBytesAllocated.addAndGet(-buffer_.capacity());
811  buffer_ = null;
812  state_ = AWAITING_CLOSE;
813  changeSelectInterests();
814  }
