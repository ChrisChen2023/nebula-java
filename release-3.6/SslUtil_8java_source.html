38  public static SSLSocketFactory getSSLSocketFactoryWithCA(CASignedSSLParam param) {
39  final String caCrtFile = param.getCaCrtFilePath();
40  final String crtFile = param.getCrtFilePath();
41  final String keyFile = param.getKeyFilePath();
42  final String password = "";
43  try {
44  //Add BouncyCastle as a Security Provider
45  Security.addProvider(new BouncyCastleProvider());
46 
47  // Load client private key
48  PEMParser reader = null;
49  Object keyObject;
50  try {
51  reader = new PEMParser(new FileReader(keyFile));
52  keyObject = reader.readObject();
53  } finally {
54  if (reader != null) {
55  reader.close();
56  }
57  }
58 
59  PEMDecryptorProvider provider =
60  new JcePEMDecryptorProviderBuilder().build(password.toCharArray());
61  JcaPEMKeyConverter keyConverter = new JcaPEMKeyConverter().setProvider("BC");
62 
63  KeyPair key;
64 
65  if (keyObject instanceof PEMEncryptedKeyPair) {
66  key = keyConverter.getKeyPair(((PEMEncryptedKeyPair) keyObject)
67  .decryptKeyPair(provider));
68  } else {
69  key = keyConverter.getKeyPair((PEMKeyPair) keyObject);
70  }
71 
72  // Load Certificate Authority (CA) certificate
73  X509CertificateHolder caCertHolder;
74  try {
75  reader = new PEMParser(new FileReader(caCrtFile));
76  caCertHolder = (X509CertificateHolder) reader.readObject();
77  } finally {
78  if (reader != null) {
79  reader.close();
80  }
81  }
82 
83  // CA certificate is used to authenticate server
84  JcaX509CertificateConverter certificateConverter =
85  new JcaX509CertificateConverter().setProvider("BC");
86  KeyStore caKeyStore = KeyStore.getInstance(KeyStore.getDefaultType());
87  X509Certificate caCert = certificateConverter.getCertificate(caCertHolder);
88  caKeyStore.load(null, null);
89  caKeyStore.setCertificateEntry("ca-certificate", caCert);
90 
91  TrustManagerFactory trustManagerFactory = TrustManagerFactory.getInstance(
92  TrustManagerFactory.getDefaultAlgorithm());
93  trustManagerFactory.init(caKeyStore);
94 
95  // Load client certificate
96  X509CertificateHolder certHolder;
97  try {
98  reader = new PEMParser(new FileReader(crtFile));
99  certHolder = (X509CertificateHolder) reader.readObject();
100  } finally {
101  if (reader != null) {
102  reader.close();
103  }
104  }
105 
106  // Client key and certificates are sent to server so it can authenticate the client
107  KeyStore clientKeyStore = KeyStore.getInstance(KeyStore.getDefaultType());
108  X509Certificate cert = certificateConverter.getCertificate(certHolder);
109  clientKeyStore.load(null, null);
110  clientKeyStore.setCertificateEntry("certificate", cert);
111  clientKeyStore.setKeyEntry("private-key", key.getPrivate(), password.toCharArray(),
112  new Certificate[]{cert});
113 
114  KeyManagerFactory keyManagerFactory = KeyManagerFactory.getInstance(
115  KeyManagerFactory.getDefaultAlgorithm());
116  keyManagerFactory.init(clientKeyStore, password.toCharArray());
117 
118  // Create SSL socket factory
119  SSLContext context = SSLContext.getInstance("TLSv1.3");
120  context.init(keyManagerFactory.getKeyManagers(),
121  trustManagerFactory.getTrustManagers(), null);
122 
123 
124  trustManagers = trustManagerFactory.getTrustManagers();
125  // Return the newly created socket factory object
126  return context.getSocketFactory();
127 
128  } catch (Exception e) {
129  LOGGER.error(e.getMessage());
130  }
131 
132  return null;
133  }
