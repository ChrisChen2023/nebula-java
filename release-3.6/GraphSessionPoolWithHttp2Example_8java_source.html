97  private static void executeForMultiThreads(SessionPool sessionPool) {
98  ExecutorService executorService = Executors.newFixedThreadPool(parallel);
99  CountDownLatch count = new CountDownLatch(parallel);
100  for (int i = 0; i < parallel; i++) {
101  executorService.submit(() -> {
102  try {
103  for (int j = 0; j < executeTimes; j++) {
104  ResultSet result = sessionPool.execute("match(v:player) return v limit 1;");
105  System.out.println(result.toString());
106  }
107  count.countDown();
108  } catch (Exception e) {
109  e.printStackTrace();
110  }
111  });
112  }
113 
114  try {
115  count.await();
116  } catch (InterruptedException e) {
117  e.printStackTrace();
118  }
119 
120  executorService.shutdown();
121  }
