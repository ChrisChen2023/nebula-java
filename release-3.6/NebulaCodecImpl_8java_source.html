52  public byte[] vertexKey(int vidLen,
53  int partitionId,
54  byte[] vertexId,
55  int tagId) {
56  if (vertexId.length > vidLen) {
57  throw new RuntimeException(
58  "The length of vid size is out of the range, expected vidLen less then " + vidLen);
59  }
60  ByteBuffer buffer = ByteBuffer.allocate(VERTEX_SIZE + vidLen);
61  buffer.order(this.byteOrder);
62  partitionId = (partitionId << 8) | VERTEX_KEY_TYPE;
63  buffer.putInt(partitionId)
64  .put(vertexId);
65  if (vertexId.length < vidLen) {
66  ByteBuffer complementVid = ByteBuffer.allocate(vidLen - vertexId.length);
67  Arrays.fill(complementVid.array(), (byte) '\0');
68  buffer.put(complementVid);
69  }
70  buffer.putInt(tagId);
71  return buffer.array();
72  }

143  buffer.order(this.byteOrder);
144  partitionId = (partitionId << 8) | EDGE_KEY_TYPE;

208  private byte[] encode(Schema schema,
209  long ver,
210  List<String> names,
211  List<Object> values)
212  throws RuntimeException {
213  if (names.size() != values.size()) {
214  throw new RuntimeException(
215  String.format("The names' size no equal with values' size, [%d] != [%d]",
216  names.size(), values.size()));
217  }
218  RowWriterImpl writer = new RowWriterImpl(genSchemaProvider(ver, schema), this.byteOrder);
219  for (int i = 0; i < names.size(); i++) {
220  writer.setValue(names.get(i), values.get(i));
221  }
222  writer.finish();
223  return writer.encodeStr();
224  }

226  private SchemaProviderImpl genSchemaProvider(long ver, Schema schema) {
227  SchemaProviderImpl schemaProvider = new SchemaProviderImpl(ver);
228  for (ColumnDef col : schema.getColumns()) {
229  ColumnTypeDef type = col.getType();
230  boolean nullable = col.isSetNullable() && col.isNullable();
231  boolean hasDefault = col.isSetDefault_value();
232  int len = type.isSetType_length() ? type.getType_length() : 0;
233  GeoShape geoShape = type.isSetGeo_shape() ? type.getGeo_shape() : GeoShape.ANY;
234  schemaProvider.addField(new String(col.getName()),
235  type.type.getValue(),
236  len,
237  nullable,
238  hasDefault ? col.getDefault_value() : null,
239  geoShape.getValue());
240  }
241  return schemaProvider;
242  }
