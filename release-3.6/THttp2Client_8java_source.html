113  public void flush() throws TTransportException {

118  // Extract request and reset buffer
119  byte[] data = requestBuffer.toByteArray();
120  requestBuffer.reset();
121  try {
122 
123  // Create request object
124  Request.Builder requestBuilder = new Request.Builder()
125  .url(url)
126  .post(RequestBody.create(MediaType.parse("application/x-thrift"), data));
127 
128  defaultHeaders.forEach(requestBuilder::header);
129  if (customHeaders != null) {
130  customHeaders.forEach(requestBuilder::header);
131  }
132 
133  Request request = requestBuilder.build();
134 
135  // Make the request
136  Response response = client.newCall(request).execute();
137  if (!response.isSuccessful()) {
138  throw new TTransportException("HTTP Response code: " + response.code());
139  }
140
