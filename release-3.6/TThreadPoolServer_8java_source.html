229  public void serve() {
230  try {
231  serverTransport_.listen();
232  } catch (TTransportException ttx) {
233  LOGGER.error("Error occurred during listening.", ttx);
234  return;
235  }
236 
237  stopped_ = false;
238  while (!stopped_) {
239  try {
240  TTransport client = serverTransport_.accept();
241  WorkerProcess wp = new WorkerProcess(client);
242  try {
243  executorService_.execute(wp);
244  } catch (RejectedExecutionException ree) {
245  LOGGER.warn("Client request was rejected by executor", ree);
246  // close the connection, we don't need it anymore
247  // if clause should always be true, but just to be on the safe side
248  if (client != null && client.isOpen()) {
249  client.close();
250  }
251  }
252  } catch (TTransportException ttx) {
253  if (!stopped_) {
254  LOGGER.warn("Transport error occurred during acceptance of message.", ttx);
255  }
256  }
257  }
258 
259  executorService_.shutdown();
260 
261  // Loop until awaitTermination finally does return without a interrupted
262  // exception. If we don't do this, then we'll shut down prematurely. We want
263  // to let the executorService clear it's task queue, closing client sockets
264  // appropriately.
265  long timeoutMS = options_.stopTimeoutUnit.toMillis(options_.stopTimeoutVal);
266  long now = System.currentTimeMillis();
267  while (timeoutMS >= 0) {
268  try {
269  executorService_.awaitTermination(timeoutMS, TimeUnit.MILLISECONDS);
270  break;
271  } catch (InterruptedException ix) {
272  long newnow = System.currentTimeMillis();
273  timeoutMS -= (newnow - now);
274  now = newnow;
275  }
276  }
277  }

299  public void run() {
300  TProcessor processor = null;
301  TTransport inputTransport = null;
302  TTransport outputTransport = null;
303  TProtocol inputProtocol = null;
304  TProtocol outputProtocol = null;
305  try {
306  processor = processorFactory_.getProcessor(client_);
307  inputTransport = inputTransportFactory_.getTransport(client_);
308  inputProtocol = inputProtocolFactory_.getProtocol(inputTransport);
309  // THeaderProtocol must be the same instance for both input and output
310  if (inputProtocol instanceof THeaderProtocol) {
311  outputProtocol = inputProtocol;
312  } else {
313  outputTransport = outputTransportFactory_.getTransport(client_);
314  outputProtocol = outputProtocolFactory_.getProtocol(outputTransport);
315  }
316  // we check stopped_ first to make sure we're not supposed to be shutting
317  // down. this is necessary for graceful shutdown.
318  TRpcConnectionContext server_ctx =
319  new TRpcConnectionContext(client_, inputProtocol, outputProtocol);
320  while (!stopped_ && processor.process(inputProtocol, outputProtocol, server_ctx)) {}
321  } catch (TTransportException ttx) {
322  // Assume the client died and continue silently
323  } catch (TException tx) {
324  LOGGER.error("Thrift error occurred during processing of message.", tx);
325  } catch (Exception x) {
326  LOGGER.error("Error occurred during processing of message.", x);
327  }
328 
329  if (inputTransport != null) {
330  inputTransport.close();
331  }
332 
333  if (outputTransport != null) {
334  outputTransport.close();
335  }
336  }
337  }
