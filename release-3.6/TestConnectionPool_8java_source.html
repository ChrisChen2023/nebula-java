121  @Test()
122  public void testGetSession() {
123  NebulaPool pool = null;
124  try {
125  NebulaPoolConfig nebulaPoolConfig = new NebulaPoolConfig();
126  nebulaPoolConfig.setMinConnSize(2);
127  nebulaPoolConfig.setMaxConnSize(4);
128  // set idle time
129  nebulaPoolConfig.setIdleTime(2000);
130  nebulaPoolConfig.setIntervalIdle(1000);
131  nebulaPoolConfig.setMinClusterHealthRate(0);
132  // set wait time
133  nebulaPoolConfig.setWaitTime(1000);
134  List<HostAddress> addresses = Collections.singletonList(
135  new HostAddress("127.0.0.1", 9671));
136  pool = new NebulaPool();
137  assert pool.init(addresses, nebulaPoolConfig);
138  int i = 0;
139  List<Session> sessions = new ArrayList<>();
140  while (i < nebulaPoolConfig.getMaxConnSize()) {
141  Session session = pool.getSession("root", "nebula", false);
142  assert (session != null);
143  ResultSet resp = session.execute("SHOW SPACES");
144  assert (resp.getErrorCode() == ErrorCode.SUCCEEDED.getValue());
145  sessions.add(session);
146  i++;
147  }
148 
149  assert (pool.getActiveConnNum() == 4);
150  assert (pool.getIdleConnNum() == 0);
151  assert (pool.getWaitersNum() == 0);
152  // All sessions are in used, so getSession failed, and the wait time is 1000ms
153  long beginTime = System.currentTimeMillis();
154  try {
155  Session session = pool.getSession("root", "nebula", false);
156  Assert.assertTrue(session == null);
157  } catch (Exception e) {
158  System.out.println(e.getMessage());
159  System.out.println("We expect must reach here: get session failed.");
160  long timeInterval = System.currentTimeMillis() - beginTime;
161  System.out.println("timeInterval is " + timeInterval);
162  Assert.assertTrue(System.currentTimeMillis() - beginTime >= 1000);
163  assert (true);
164  }
165 
166  // Test reuse session after the sessions have been release
167  for (Session s : sessions) {
168  s.release();
169  }
170  Assert.assertEquals(0, pool.getActiveConnNum());
171  Assert.assertEquals(4, pool.getIdleConnNum());
172  Session session = pool.getSession("root", "nebula", false);
173  Assert.assertNotNull(session);
174  Assert.assertEquals(1, pool.getActiveConnNum());
175  Assert.assertEquals(3, pool.getIdleConnNum());
176 
177  // test idletime connection is closed
178  session.release();
179  TimeUnit.SECONDS.sleep(3);
180  Assert.assertEquals(2, pool.getIdleConnNum());
181  Assert.assertEquals(0, pool.getActiveConnNum());
182 
183  // Test multi release
184  session.release();
185  session.release();
186  } catch (Exception e) {
187  e.printStackTrace();
188  assert (false);
189  } finally {
190  if (pool != null) {
191  pool.close();
192  }
193  }
