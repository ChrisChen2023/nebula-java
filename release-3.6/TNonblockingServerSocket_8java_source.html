66  public TNonblockingServerSocket(int port, int clientTimeout, int backlog)
67  throws TTransportException {
68  port_ = port;
69  clientTimeout_ = clientTimeout;
70  try {
71  serverSocketChannel = ServerSocketChannel.open();
72  serverSocketChannel.configureBlocking(false);
73 
74  // Make server socket
75  serverSocket_ = serverSocketChannel.socket();
76  // Prevent 2MSL delay problem on server restarts
77  serverSocket_.setReuseAddress(true);
78  // Bind to listening port
79  serverSocket_.bind(new InetSocketAddress(port_), backlog);
80  } catch (IOException ioe) {
81  serverSocket_ = null;
82  throw new TTransportException("Could not create ServerSocket on port " + port + ".");
83  }
84  }
