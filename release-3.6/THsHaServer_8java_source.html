227  @Override
228  public void serve() {
229  if (!startInvokerPool()) {
230  return;
231  }

234  if (!startListening()) {
235  return;
236  }
237 
238  // start the selector, or exit
239  if (!startSelectorThread()) {
240  return;
241  }
242 
243  // this will block while we serve
244  joinSelector();
245 
246  gracefullyShutdownInvokerPool();
247 
248  // do a little cleanup
249  stopListening();
250 
251  // ungracefully shut down the invoker pool?
252  }

254  protected boolean startInvokerPool() {
255  // start the invoker pool
256  BlockingQueue<Runnable> queue =
257  (options_.hsHaQueueSize == 0)
258  ? new SynchronousQueue<Runnable>()
259  : new LinkedBlockingQueue<Runnable>(options_.hsHaQueueSize);
260  invoker =
261  new ThreadPoolExecutor(
262  options_.minHsHaWorkerThreads,
263  options_.maxHsHaWorkerThreads,
264  options_.stopTimeoutVal,
265  options_.stopTimeoutUnit,
266  queue,
267  new TThreadFactoryImpl("THsHaServer-Invoker"));
268 
269  return true;
270  }
271 
272  protected void gracefullyShutdownInvokerPool() {
273  // try to gracefully shut down the executor service
274  invoker.shutdown();
275 
276  // Loop until awaitTermination finally does return without a interrupted
277  // exception. If we don't do this, then we'll shut down prematurely. We want
278  // to let the executorService clear it's task queue, closing client sockets
279  // appropriately.
280  long timeoutMS = 10000;
281  long now = System.currentTimeMillis();
282  while (timeoutMS >= 0) {
283  try {
284  invoker.awaitTermination(timeoutMS, TimeUnit.MILLISECONDS);
285  break;
286  } catch (InterruptedException ix) {
287  long newnow = System.currentTimeMillis();
288  timeoutMS -= (newnow - now);
289  now = newnow;
290  }
291  }
292  }
