33  private void resetProtocol() throws TProtocolException {
34  // We guarantee trans_ to be of type THeaderTransport, parent class does not
35  if (proto != null && protoId == ((THeaderTransport) trans_).getProtocolId()) {
36  return;
37  }
38 
39  protoId = ((THeaderTransport) trans_).getProtocolId();
40  switch (protoId) {
41  case THeaderTransport.T_BINARY_PROTOCOL:
42  proto = new TBinaryProtocol(trans_, true, true);
43  break;
44  case THeaderTransport.T_COMPACT_PROTOCOL:
45  proto = new TCompactProtocol(trans_);
46  break;
47  default:
48  throw new TProtocolException("Unknown protocol id: " + protoId);
49  }
50  }

203  public void notifyEndpoint(String msg) throws TException {
